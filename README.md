  ID NUMBER(5,0) NOT NULL,

  CREATE TABLE PROJECTS(
  ID NUMBER(5,0) NOT NULL,
  NAME VARCHAR2(200 BYTE),
  CLIENT_ID NUMBER(5,0) NOT NULL,
  CONSTRAINT PROJECTS_PK PRIMARY KEY(ID));
  
  CREATE TABLE CLIENTS(
  ID NUMBER(5,0) NOT NULL,
  NAME VARCHAR2 (200 byte),
  CONSTRAINT CLIENTS_PK PRIMARY KEY(ID));
